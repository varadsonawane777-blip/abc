#include <stdio.h>
#define MAX 100

int binarySearch(int arr[], int n, int key) {
    int low = 0, high = n - 1, mid;

    while (low <= high) {
        mid = (low + high) / 2;

        if (arr[mid] == key)
            return mid;
        else if (arr[mid] < key)
            low = mid + 1;
        else
            high = mid - 1;
    }
    return -1;
}

int main() {
    int rollNumbers[MAX];
    int n, key, result;

    printf("Enter number of students who attended the training program: ");
    scanf("%d", &n);

    printf("Enter roll numbers of students (in sorted order):\n");
    for (int i = 0; i < n; i++) {
        scanf("%d", &rollNumbers[i]);
    }

    printf("Enter roll number to search: ");
    scanf("%d", &key);

    result = binarySearch(rollNumbers, n, key);

    if (result == -1)
        printf("Student with roll number %d did NOT attend the training program.\n", key);
    else
        printf("Student with roll number %d attended the training program.\n", key);

    return 0;
}
