#include <stdio.h>

#define MAX 100

int linearSearch(int arr[], int n, int key) {
    for (int i = 0; i < n; i++) {
        if (arr[i] == key)
            return i;
    }
    return -1;
}

int main() {
    int rollNumbers[MAX];
    int n, key, result;

    printf("Enter number of students who attended the training program: ");
    scanf("%d", &n);

    printf("Enter roll numbers of students (in random order):\n");
    for (int i = 0; i < n; i++) {
        scanf("%d", &rollNumbers[i]);
    }

    printf("Enter roll number to search: ");
    scanf("%d", &key);

    result = linearSearch(rollNumbers, n, key);

    if (result == -1)
        printf("Student with roll number %d did NOT attend the training program.\n", key);
    else
        printf("Student with roll number %d attended the training program.\n", key);

    return 0;
}
